\section{Information}

\begin{definitionbox}[Information]
Information refers to data or knowledge that is meaningful and has value in making decisions or solving problems.
\end{definitionbox}

\subsection{Importance of Information Theory}

\subsubsection{Why Information Theory Matters}
\begin{itemize}
    \item It helps us \textbf{quantify} information.
    \item It helps us \textbf{transmit} information efficiently.
    \item It helps us \textbf{optimize the delivery} of information in communication, computing, and biological systems.
\end{itemize}

\subsubsection{Real-Life Relevance}
\begin{itemize}
    \item Modern systems such as the \textbf{internet}, \textbf{mobile networks}, and \textbf{social media} rely on efficient data compression and error correction.
    \item These ideas ensure messages are transmitted \textbf{accurately} and \textbf{efficiently}.
    \item Information theory also helps reduce \textbf{noise interference} in signals.
    \item This improves reliability in applications such as \textbf{streaming}, \textbf{medical diagnostics}, and \textbf{genomic analysis}.
\end{itemize}

\subsection{Importance in Communication}

\subsubsection{Key Points}
\begin{itemize}
    \item Information theory provides mathematical frameworks to optimize \textbf{data transmission} and \textbf{storage}.
    \item It helps determine the maximum amount of information a channel can handle, that is, \textbf{channel capacity}.
    \item It helps determine how to encode messages so that transmission errors can be reduced.
\end{itemize}

\subsubsection{Communication Insight}
\begin{itemize}
    \item Cellular networks and internet protocols use \textbf{error correction} and \textbf{data compression}.
    \item These make communication \textbf{efficient} and \textbf{reliable}.
    \item Even in noisy environments, users can still experience:
    \begin{itemize}
        \item clear calls,
        \item fast internet speeds,
        \item accurate data exchange.
    \end{itemize}
\end{itemize}

\noindent This is why information theory is fundamental to seamless global communication.

\section{Discrete Source}

\begin{definitionbox}[Discrete Source]
A discrete source is a system that generates a \textbf{finite set of symbols or values} from a \textbf{predefined set of alphabets}. These symbols are discrete and distinct, such as letters, numbers, or pixels.
\end{definitionbox}

\subsubsection{Examples of Discrete Sources}
\begin{itemize}
    \item \textbf{Text:} A keyboard generates letters, numbers and special characters.
    \item \textbf{Audio:} Digital audio files sample sound at specific intervals, producing discrete data points.
    \item \textbf{Image/Video:} Images are composed of pixels with distinct color values, while videos are sequences of discrete image frames.
\end{itemize}

\subsection{Source Types}
\subsubsection{Memoryless Source}

\begin{definitionbox}[Memoryless Source]
A discrete source is memoryless if the symbols it generates are \textbf{independent} of each other.
\end{definitionbox}

\begin{itemize}
    \item The current symbol has \textbf{no dependence} on past symbols.
    \item Example: flipping a fair coin repeatedly produces independent outcomes $(H \text{ or } T)$.
    \item Such sources simplify analysis because each symbol can be treated individually without considering historical context.
    \item Example:A random number generator producing digits $(0\text{--}9)$ where each number is equally likely and independent of the previous numbers.
\end{itemize}

\subsubsection{Memory Source}

\begin{definitionbox}[Memory Source]
A discrete source is a memory source if the generation of a symbol \textbf{depends} on one or more \textbf{previous symbols}.
\end{definitionbox}

\begin{itemize}
    \item Such sources have \textbf{statistical dependency} between symbols.
    \item This dependency can be modeled using \textbf{Markov processes} or \textbf{conditional probabilities}.
    \item They are common in \textbf{natural language and encoded data streams}, where rules and structure influence symbol generation.
    \item Example: In English text, the letter `q' is almost always followed by `u', showing a strong dependency between these symbols.
\end{itemize}

\section{Self Information}

\begin{definitionbox}[Self Information]
Self information measures the \textbf{amount of uncertainty or surprise} associated with the occurrence of a specific event. It quantifies how much \textbf{information is gained} when an event occurs.
\end{definitionbox}

\begin{formulabox}[Self Information Formula]
Consider a discrete random variable $X$ with possible outcomes $x_i$, $i = 1, 2, 3, \dots, n$. Self information of the event $X = x_i$ is defined as
\[
I(x_i) = \log \frac{1}{P(x_i)} = -\log P(x_i).
\]
\begin{itemize}
    \item $I(x_i)$ = self information of the event $x_i$
    \item $P(x_i)$ = probability of the event $x_i$
\end{itemize}
\end{formulabox}

\subsection{Properties of Self Information}
\begin{enumerate}
    \item If we are absolutely certain of the outcome of an event even before it occurs, there is no information gained.
    \[
    I(x_i) = 0, \quad \text{if } P_i = 1
    \]
    This means \textbf{certainty carries no message information}.
    \item The occurrence of an event either provides some or no information, but never brings about a loss of information.
    \[
    I(x_i) \geq 0, \quad \text{for } 0 \leq P(x_i) \leq 1
    \]
    That implies \textbf{self information cannot be negative}.
    \item The \textbf{less probable an event is, the more information we gain} when it occurs.
    \[
    I(x_1) > I(x_2) \quad \text{if } P_1 < P_2
    \]
    \item For statistically independent events, the total information obtained from the joint occurrence is equal to the sum of the individual informations.
    \[
    I(x_1, x_2) = I(x_1) + I(x_2).
    \]
    This means that \textbf{information is additive for independent events}.

    Equivalently,
    \[
    I_A + I_B = \log \frac{1}{P_A} + \log \frac{1}{P_B}
    = \log \frac{1}{P_A P_B}
    = \log \frac{1}{P(AB)}.
    \]

    This follows because for independent events,
    \[
    P(AB) = P_A \cdot P_B.
    \]

    Thus, the information of the joint event can be written as
    \[
    I(AB) = \log \frac{1}{P(AB)},
    \]
    which matches the sum of the individual informations. This implies that \textbf{independent events do not share information, so their informations simply add up}.
\end{enumerate}

\subsection{Physical significance / insight of self information}
Consider a set of events $X = \{x_1, x_2, x_3\}$ with probability
\[
P = \{0.5, 0.25, 0.25\}.
\]

\textbf{Step 1:} Self information for the first event:
\[
I_{x_1} = \log_2 \frac{1}{0.5} = -\log_2 0.5 = 1 \text{ bit}
\]

\textbf{Step 2:} For the other two events:
\[
I_{x_2} = I_{x_3} = \log_2 \frac{1}{0.25} = -\log_2 0.25 = 2 \text{ bit}
\]

\begin{infobox}[Observation]
\begin{itemize}
    \item Higher probability requires fewer bits.
    \item Lower probability requires more bits.
    \item Lower probability therefore means a higher degree of uncertainty.
\end{itemize}
\end{infobox}

\section{Entropy}

\begin{definitionbox}[Entropy]
In information theory, entropy is the measurement of the average amount of information, that is, the number of bits of uncertainty, in a set of possible outcomes per symbol.
\end{definitionbox}

\subsection{Interpretation}
\begin{itemize}
    \item It measures the \textbf{unpredictability} of a random variable.
    \item It represents the minimum number of bits required to encode the information generated by a source.
\end{itemize}

\begin{formulabox}[Entropy of an Event]
\[
H(x_i) = -P_i \log_2 P_i.
\]
\begin{itemize}
    \item $H(x_i)$ = entropy contribution of the event $x_i$
    \item $P_i$ = probability of the $i$th event
\end{itemize}
\end{formulabox}

\subsection{Why This Form Appears}
\begin{itemize}
    \item Self information of an event is $\left(-\log_2 P_i\right)$.
    \item Entropy is an \textbf{average} information measure.
    \item So self information is weighted by its probability $(P_i)$.
\end{itemize}

\begin{formulabox}[Entropy of a Random Variable]
\[
H(X) = -\sum_{i=1}^{n} P_i \log_2 P_i
\quad \text{bit/symbol.}
\]
\begin{itemize}
    \item $H(X)$ = average entropy of the random variable $X$
    \item $n$ = total number of possible outcomes
    \item $P_i$ = probability of the $i$th outcome
\end{itemize}
\end{formulabox}

\begin{questionanswerbox}[Example]
A source generates information with probabilities
\[
P_1 = 0.1,\quad P_2 = 0.2,\quad P_3 = 0.3,\quad P_4 = 0.4.
\]
Find the entropy of the system and what percentage of maximum possible information is being generated by this source.

\textbf{Step 1: Calculate entropy}
\[
H(X) = -\sum_{i=1}^{n} P_i \log_2 P_i
\]
\[
= -\Bigl(0.1\log_2(0.1) + 0.2\log_2(0.2) + 0.3\log_2(0.3) + 0.4\log_2(0.4)\Bigr)
\]
\[
= 1.8464 \approx 2 \text{ bits/symbol.}
\]

\textbf{Step 2: Calculate maximum entropy}
\[
H_{\max} = \log_2 4 = 2.
\]

\textbf{Step 3: Calculate efficiency}
\[
\eta = \frac{H}{H_{\max}} \times 100
= \frac{1.8464}{2}\times 100
= 92.32\%.
\]
\end{questionanswerbox}

\begin{questionanswerbox}[Question]
\textbf{Q:} Why is \texorpdfstring{$P_i$}{Pi} multiplied with self information in the entropy equation?

\textbf{A:}
\begin{itemize}
    \item Entropy is an \emph{average} measure of information.
    \item So the self information term $\left(-\log_2 P_i\right)$ must be weighted by its probability $P_i$.
    \item This keeps highly probable events contributing more frequently and less probable events contributing less frequently to the final average.
    \item It also keeps the entropy value bounded properly for a valid probability distribution.
            For example- the probability must be in between 0 and 1. If - 
            \begin{align*}
                P = 0.1 &\Rightarrow I = -\log_2(0.1) = 3.32 \text{ bits} \text{ (more than 1 bit)} \\
                P = 0.5 &\Rightarrow I = -\log_2(0.6) = 0.74 \text{ bit} \\
                P = 0.9 &\Rightarrow I = -\log_2(0.9) = 0.152 \text{ bits}
            \end{align*}
            To maintain entropy value below 1 bit/symbol, we need to multiply the self information with the probability $P_i$.
\end{itemize}
\end{questionanswerbox}

\subsection{Significance of Entropy}

\begin{itemize}
    \item \textbf{Measures uncertainty:} quantifies unpredictability in a source, providing insight into the information content.
    \item \textbf{Data compression:} sets the theoretical limits for lossless compression, helping design efficient encoding systems.
    \item \textbf{Error detection/correction:} guides the design of coding techniques to handle noise in communication channels.
    \item \textbf{System efficiency:} helps evaluate and optimize the performance of communication and storage systems.
    \item \textbf{Applications in AI/ML:} used in decision trees and deep learning for feature selection and model optimization.
\end{itemize}

\subsection{Relation between Hartley, Nats and Bits}

\subsubsection{Units of Self Information}
\begin{itemize}
    \item Self information in Hartley:
    \[
    I = \log_{10}\frac{1}{P}
    \]
    \item Self information in nats:
    \[
    I = \log_e \frac{1}{P}
    \]
    \item Self information in bits:
    \[
    I = \log_2 \frac{1}{P}
    \]
\end{itemize}

\begin{formulabox}[Hartley to Bits]
\[
1 \text{ Hartley}
= \frac{I}{-\log_{10} P}
= \frac{-\log_2 P}{-\log_{10} P}
= \frac{\log 10}{\log 2}
= \log_2 10
= 3.3219 \text{ bits.}
\]
\end{formulabox}

\begin{formulabox}[Hartley to Nats]
\[
1 \text{ Hartley}
= \frac{I}{-\log_{10} P}
= \frac{-\log_e P}{-\log_{10} P}
= \frac{\log 10}{\log e}
= \log_e 10
= 2.302 \text{ nats.}
\]
\end{formulabox}

\begin{formulabox}[Nats to Bits]
\[
1 \text{ Nat}
= \frac{I}{-\log_e P}
= \frac{-\log_2 P}{-\log_e P}
= \frac{\log e}{\log 2}
= \log_2 e
= \frac{1}{\ln 2}
= 1.442 \text{ bits.}
\]
\end{formulabox}

\begin{questionanswerbox}[Example]
\textbf{Q:} If there are $100$ equiprobable outcomes, calculate the corresponding values in nats, bits and Hartley.

\textbf{A:} Given $N = 100$, for equiprobable outcomes:
\[
P(x_i) = \frac{1}{100}
\]

\textbf{Using entropy formula:}
\[
H(X) = -\sum_{i=1}^{100} \frac{1}{100} \log \frac{1}{100}
\]

Since all terms are identical:
\[
H(X) = -100 \cdot \frac{1}{100} \log \frac{1}{100}
\]

\[
H(X) = -\log \frac{1}{100}
\]

Now simplify:
\[
H(X) = \log 100
\]

\textbf{Concept:}
\begin{itemize}
    \item Bits, nats, and Hartley are \textbf{units of information} (log base), used for both $I(x)$ and $H(X)$.
    \item The question refers to a \textbf{set of outcomes}, so we use \textbf{entropy}:
    \[
    H(X) = -\sum P(x_i)\log P(x_i)
    \]
    \item For equiprobable case:
    \[
    I(x_i) = -\log \frac{1}{100} = \log 100, \quad H(X) = \log 100
    \]
    \item Hence, \textbf{self-information = entropy} in this case.
    \item Units depend on log base:
    \[
    \log_2 \rightarrow \text{bits}, \quad
    \log_e \rightarrow \text{nats}, \quad
    \log_{10} \rightarrow \text{Hartley}
    \]
\end{itemize}

\textbf{Note:} Bits, nats and Hartley are units of information (log base).

\begin{align*}
\text{In bits, } H(X) &= \log_2(100) = 6.64 \\
\text{In nats, } H(X) &= \log_e(100) = 4.605 \\
\text{In Hartley, } H(X) &= \log_{10}(100) = 2
\end{align*}
\end{questionanswerbox}

\subsection{Equal Probability Leads to High Entropy}

Consider a BPSK output
\[
x =
\begin{cases}
1, & \text{probability } p \\
0, & \text{probability } 1-p
\end{cases}
\]

\[
H(x) = -\sum_{x=0}^{1} P(x)\log_2 P(x)
\]
\[
= -p\log_2 p - (1-p)\log_2(1-p).
\]
\begin{itemize}
    \item $p$ = probability of symbol $1$
    \item $1-p$ = probability of symbol $0$
    \item $H(x)$ = entropy of the binary source
\end{itemize}

For equal probability $p=0.5$ and $1-p=0.5$,
\[
H(x) = -0.5\log_2 0.5 - 0.5\log_2 0.5 = 1 \text{ bit/symbol.}
\]

\begin{figure}[H]
    \centering
    \begin{tikzpicture}[scale=0.9]
        \draw[->] (0,0) -- (4.6,0) node[right] {$P$};
        \draw[->] (0,0) -- (0,3.1) node[above] {$H(P)$};
        \draw[smooth,domain=0.05:0.95,samples=120,thick,color=chaptercolor]
            plot (\x*4,{(-\x*ln(\x)/ln(2) - (1-\x)*ln(1-\x)/ln(2))*2.8});
        \draw[dashed] (2,0) -- (2,2.8);
        \node[below] at (0,0) {$0$};
        \node[below] at (2,0) {$0.5$};
        \node[below] at (4,0) {$1$};
        \node[left] at (0,2.8) {$1$};
    \end{tikzpicture}
    \caption{Entropy of a binary source versus probability}
\end{figure}

From the figure, it is clear that for $p=0$ or $p=1$, entropy $H(x)$ is zero, but for $p=0.5$, entropy is maximum. Equal probability leads to the highest uncertainty, so it also leads to the highest entropy.

\begin{questionanswerbox}[Example]
Calculate the entropy of the QPSK transmitted signal below:

\[
00\;\;01\;\;10\;\;10\;\;11\;\;11\;\;01\;\;01\;\;10\;\;00\;\;10\;\;10\;\;11\;\;01\;\;10
\]

QPSK has four distinct $(00, 01, 10, 11)$ bit combinations. Thus, $2^n = 4 \Rightarrow n = 2$. So, we have to take two bits for each symbol.

\renewcommand{\arraystretch}{1.3}
\setlength{\tabcolsep}{8pt}
\begin{center}
\begin{tabular}{|C{2.8cm}|C{2.2cm}|}
    \hline
    \textbf{Symbol} & \textbf{Times} \\ \hline
    00 & 2 \\ \hline
    01 & 4 \\ \hline
    10 & 6 \\ \hline
    11 & 3 \\ \hline
\end{tabular}
\end{center}

\[
P_{00} = \frac{2}{15}, \qquad
P_{01} = \frac{4}{15}, \qquad
P_{10} = \frac{6}{15}, \qquad
P_{11} = \frac{3}{15}
\]

\[
H(X) = -\left[
P_{00}\log_2 P_{00}
+ P_{01}\log_2 P_{01}
+ P_{10}\log_2 P_{10}
+ P_{11}\log_2 P_{11}
\right]
\]
\[
= -\left[
\frac{2}{15}\log_2\frac{2}{15}
+ \frac{4}{15}\log_2\frac{4}{15}
+ \frac{6}{15}\log_2\frac{6}{15}
+ \frac{3}{15}\log_2\frac{3}{15}
\right]
\]
\[
= 1.889 \approx 2 \text{ bits/symbol.}
\]

\textbf{Physical significance:} Thus, we need at least $2$ bits to transmit the signal.
\end{questionanswerbox}

\section{Joint Entropy}

\begin{definitionbox}[Joint Entropy]
The joint entropy $H(X,Y)$ of a pair of random variables $(X,Y)$ with joint distribution $P(x,y)$ is the average uncertainty associated with the pair of outcomes taken together.
\end{definitionbox}

\begin{formulabox}[Joint Entropy Formula]
\[
H(X,Y) = -\sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(x,y)
\]
\[
= -E\log_2 P(x,y)
\]
\begin{itemize}
    \item $H(X,Y)$ = joint entropy of the variables $X$ and $Y$
    \item $P(x,y)$ = joint probability of the pair $(x,y)$
    \item $E[\cdot]$ = expectation
\end{itemize}
\end{formulabox}

\begin{questionanswerbox}[Example]
Two variables are considered for a student:
\begin{itemize}
    \item $X$ = study hours
    \item $Y$ = attendance
\end{itemize}

Their joint probabilities are listed below. Show that the variables are dependent on each other.

\renewcommand{\arraystretch}{1.3}
\setlength{\tabcolsep}{8pt}
\begin{center}
\begin{tabular}{|C{2.8cm}|C{2.8cm}|C{2.8cm}|}
    \hline
    \textbf{X} & \textbf{Y} & $\mathbf{P(X,Y)}$ \\ \hline
    Low  & Low  & 0.20 \\ \hline
    Low  & High & 0.10 \\ \hline
    High & Low  & 0.15 \\ \hline
    High & High & 0.55 \\ \hline
\end{tabular}
\end{center}

\textbf{Step 1: Joint entropy contribution of each case}
\begin{align*}
H_{LL}(X,Y) &= -0.2\log_2(0.2) = 0.46 \\
H_{LH}(X,Y) &= -0.1\log_2(0.1) = 0.33 \\
H_{HL}(X,Y) &= -0.15\log_2(0.15) = 0.41 \\
H_{HH}(X,Y) &= -0.55\log_2(0.55) = 0.4743
\end{align*}

\[
H(X,Y) = 0.46 + 0.33 + 0.41 + 0.4743 = 1.6743
\]

\textbf{Step 2: Marginal entropy for study hours}
\[
P_L = 0.2 + 0.1 = 0.3, \qquad P_H = 0.15 + 0.55 = 0.7
\]
\[
H(X) = -\left[0.3\log_2(0.3) + 0.7\log_2(0.7)\right] = 0.881
\]

\textbf{Step 3: Marginal entropy for attendance}
\[
P_L = 0.2 + 0.15 = 0.35, \qquad P_H = 0.1 + 0.55 = 0.65
\]
\[
H(Y) = -\left[0.35\log_2(0.35) + 0.65\log_2(0.65)\right] = 0.934
\]
\end{questionanswerbox}

\begin{infobox}[Conclusion]
\[
H(X) + H(Y) = 0.881 + 0.934 = 1.815
\]
Since $H(X) + H(Y) > H(X,Y)$, the variables $X$ and $Y$ are dependent. We can also say that the variables contain mutual information.
\end{infobox}

\section{Mutual Information}

\begin{definitionbox}[Mutual Information]
Mutual information measures the amount of information that one random variable $X$ provides about another random variable $Y$. It quantifies the reduction in uncertainty about one variable when the other is known.
\end{definitionbox}

\begin{formulabox}[Mutual Information Formula]
\[
I(X;Y) = H(X) + H(Y) - H(X,Y)
\]
\[
\text{or equivalently,} \qquad I(X;Y) = H(X) - H(X \mid Y)
\]
\begin{itemize}
    \item $I(X;Y)$ = mutual information between $X$ and $Y$
    \item $H(X)$, $H(Y)$ = marginal entropies
    \item $H(X,Y)$ = joint entropy
    \item $H(X \mid Y)$ = conditional entropy of $X$ given $Y$
\end{itemize}
\end{formulabox}

\begin{questionanswerbox}[Example]
Calculate the joint entropy for the following data and also find the mutual information.

\renewcommand{\arraystretch}{1.8}
\setlength{\tabcolsep}{8pt}
\begin{center}
\begin{tabular}{|C{3.4cm}|C{2.4cm}|C{2.4cm}|}
    \hline
    \textbf{Y $\backslash$ X} & 0 & 1 \\ \hline
    $0$ & $\frac{1}{2}$ & $\frac{1}{4}$ \\ \hline
    $1$ & $\frac{1}{8}$ & $\frac{1}{8}$ \\ \hline
\end{tabular}
\end{center}

\textbf{Step 1: Joint entropy}
\begin{align*}
H(X,Y) &= -\left[\frac{1}{2}\log_2\frac{1}{2} + \frac{1}{4}\log_2\frac{1}{4}
+ \frac{1}{8}\log_2\frac{1}{8} + \frac{1}{8}\log_2\frac{1}{8}\right] \\
&= 1.75
\end{align*}

\textbf{Step 2: Entropy of $Y$}
\[
P(Y=0) = \frac{1}{2} + \frac{1}{4} = \frac{3}{4}, \qquad
P(Y=1) = \frac{1}{8} + \frac{1}{8} = \frac{1}{4}
\]
\[
H(Y) = \frac{3}{4}\log_2\frac{4}{3} + \frac{1}{4}\log_2 4 = 0.811
\]

\textbf{Step 3: Entropy of $X$}
\[
P(X=0) = \frac{1}{2} + \frac{1}{8} = \frac{5}{8}, \qquad
P(X=1) = \frac{1}{4} + \frac{1}{8} = \frac{3}{8}
\]
\[
H(X) = \frac{5}{8}\log_2\frac{8}{5} + \frac{3}{8}\log_2\frac{8}{3} = 0.954
\]

\textbf{Step 4: Conditional entropies}
\[
H(X \mid Y) = H(X,Y) - H(Y) = 1.75 - 0.811 = 0.939
\]
\[
H(Y \mid X) = H(X,Y) - H(X) = 1.75 - 0.954 = 0.796
\]

\textbf{Step 5: Mutual information}
\[
I(X;Y) = H(X) + H(Y) - H(X,Y) = 0.954 + 0.811 - 1.75 = 0.015
\]
\end{questionanswerbox}

\section{Conditional Entropy}

\begin{definitionbox}[Conditional Entropy]
Conditional entropy measures the amount of uncertainty remaining in a random variable when another related variable is already known. It tells us how much additional information is still needed to describe X when Y is already provided.
\end{definitionbox}

\subsection{Simple Intuition}
Consider the set $\{1,2,3,4,5,6\}$. If we only know whether the outcome is odd or even, some uncertainty still remains.
\begin{itemize}
    \item If the outcome is \textbf{odd}, the possible values are $\{1,3,5\}$.
    \item If the outcome is \textbf{even}, the possible values are $\{2,4,6\}$.
\end{itemize}
This remaining uncertainty is the idea captured by conditional entropy.

\begin{formulabox}[Conditional Entropy Formula]
\[
H(Y \mid X) = -\sum_{x \in X} P(x)\sum_{y \in Y} P(y \mid x)\log_2 P(y \mid x)
\]
\[
= -\sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(y \mid x)
\]
\begin{itemize}
    \item $H(Y \mid X)$ = uncertainty of $Y$ after $X$ is known
    \item $P(y \mid x)$ = conditional probability of $Y=y$ given $X=x$
    \item $P(x,y)$ = joint probability of $X$ and $Y$
\end{itemize}
\end{formulabox}

\begin{formulabox}[Useful Relation for the Derivation]
\[
P(y \mid x)=\frac{P(x,y)}{P(x)}
\]
\[
\log(AB)=\log A+\log B
\]
\begin{itemize}
    \item The first relation rewrites conditional probability in terms of joint and marginal probability.
    \item The logarithm identity is used when expanding \(\log_2\bigl(P(x)P(y \mid x)\bigr)\).
\end{itemize}
\end{formulabox}

\subsection{Relation with Joint Entropy}
Starting from the joint entropy,
\begin{align*}
H(X,Y) &= -\sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(x,y) \\
&= -\sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2\bigl(P(x)P(y \mid x)\bigr) \\
&= -\sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(x)
- \sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(y \mid x) \\
&= -\sum_{x \in X} P(x)\log_2 P(x)
- \sum_{x \in X}\sum_{y \in Y} P(x,y)\log_2 P(y \mid x)
\end{align*}

Hence,
\[
H(X,Y) = H(X) + H(Y \mid X)
\]

Similarly,
\[
H(X,Y) = H(Y) + H(X \mid Y)
\]

\section{Relative Entropy}

\begin{definitionbox}[Relative Entropy]
Relative entropy, also called Kullback-Leibler divergence, measures how different an estimated probability distribution is from the true distribution.
\end{definitionbox}

\begin{infobox}[What Relative Entropy Means]
\begin{itemize}
    \item Relative entropy compares two probability mass functions: \(P(x)\) and \(Q(x)\).
    \item \(P(x)\) represents the true or actual distribution.
    \item \(Q(x)\) represents the estimated or assumed distribution.
    \item It tells us how far the estimate \(Q(x)\) is from the true distribution \(P(x)\).
    \item A larger value of \(D(P \parallel Q)\) means the deviation from the actual distribution is higher.
    \item If the two distributions are identical, then the relative entropy becomes zero.
\end{itemize}
\end{infobox}

\begin{formulabox}[Kullback-Leibler Divergence]
\[
D(P \parallel Q) = \sum_{x \in X} P(x)\log_2 \frac{P(x)}{Q(x)}
\]
\begin{itemize}
    \item $P(x)$ = true probability mass function
    \item $Q(x)$ = estimated probability mass function
    \item $D(P \parallel Q)$ = relative entropy from $Q$ to $P$
\end{itemize}
\end{formulabox}

\begin{infobox}[Basic Points]
\begin{itemize}
    \item If $P(x) = Q(x)$, then $D(P \parallel Q) = 0$.
        \begin{figure}[H]
        \centering
        \begin{tikzpicture}[x=1.7cm,y=2.1cm]
            \draw[->, thick] (0,0) -- (0,1.45);
            \draw[->, thick] (0,0) -- (3.4,0);

            \draw[thick] (0.45,0) rectangle (0.95,1.1);
            \draw[thick] (1.55,0) rectangle (2.05,0.8);
            \draw[thick] (2.65,0) rectangle (3.15,0.68);

            \node[above] at (0.7,1.1) {$P$};
            \node[above] at (1.8,0.8) {$A$};
            \node[above] at (2.9,0.68) {$B$};
        \end{tikzpicture}
        \caption{Relative entropy illustration comparing the true distribution $P$ with estimates $A$ and $B$}
        \end{figure}
    \item In the illustration:
    \[
    D(P \parallel P) = 0, \qquad D(P \parallel A) = 0.25, \qquad D(P \parallel B) = 1.25
    \]
    \item Higher relative entropy means the estimated distribution deviates more from the actual distribution.
\end{itemize}
\end{infobox}


\begin{questionanswerbox}[Example]
Consider a scenario where the true distribution is
\[
P_{\text{dog}} = 1, \qquad P_{\text{cat}} = 0
\]

For model $A$,
\[
P'_{\text{dog}} = 0.1, \qquad P'_{\text{cat}} = 0.9
\]

For model $B$,
\[
P'_{\text{dog}} = 0.9, \qquad P'_{\text{cat}} = 0.1
\]

Find the relative entropy using KL divergence.

\textbf{For model $A$:}
\[
D(P \parallel P'_A)
= P_{\text{dog}}\log_2\frac{P_{\text{dog}}}{P'_{\text{dog}}}
+ P_{\text{cat}}\log_2\frac{P_{\text{cat}}}{P'_{\text{cat}}}
\]
\[
= 1\log_2\frac{1}{0.1} + 0\cdot\log_2\frac{0}{0.9}
= \log_2 10
= 3.32 \text{ unit}
\]

\textbf{For model $B$:}
\[
D(P \parallel P'_B)
= P_{\text{dog}}\log_2\frac{P_{\text{dog}}}{P'_{\text{dog}}}
+ P_{\text{cat}}\log_2\frac{P_{\text{cat}}}{P'_{\text{cat}}}
\]
\[
= 1\log_2\frac{1}{0.9} + 0\cdot\log_2\frac{0}{0.1}
= 0.15 \text{ unit}
\]

So model $B$ is better because it has lower KL divergence.
\end{questionanswerbox}

\subsection{Mutual Information as Relative Entropy}

\begin{formulabox}[Mutual Information from Relative Entropy]
For two dependent random variables $(X,Y)$, mutual information can be written as the relative entropy between the joint distribution $P(x,y)$ and the product of the marginals $P(x)P(y)$:

\[
I(X;Y) = \sum_{x,y} P(x,y)\log_2 \frac{P(x,y)}{P(x)P(y)}
= D\bigl(P(x,y)\,\|\,P(x)P(y)\bigr)
\]
\begin{itemize}
    \item $I(X;Y)$ = mutual information between $X$ and $Y$
    \item $P(x,y)$ = joint probability distribution of $X$ and $Y$
    \item $P(x)$, $P(y)$ = marginal probability distributions of $X$ and $Y$
    \item $P(x)P(y)$ = product of marginals, representing the independent case
    \item $D\bigl(P(x,y)\,\|\,P(x)P(y)\bigr)$ = relative entropy between the joint distribution and the product of marginals
\end{itemize}
\end{formulabox}

\begin{infobox}[Simple Meaning]
\begin{itemize}
    \item If $X$ and $Y$ are independent, then \(P(x,y)=P(x)P(y)\).
    \item Then the two distributions inside the KL divergence are the same, so the value becomes zero.
    \item If they are dependent, then \(P(x,y)\) differs from \(P(x)P(y)\), so the value becomes positive.
    \item So mutual information measures how far the pair $(X,Y)$ is from being independent.
\end{itemize}
\end{infobox}

\begin{align*}
I(X;Y)
&= \sum_{x,y} P(x,y)\log_2 \frac{P(x,y)}{P(x)P(y)}\\
&= \sum_{x,y} P(x,y)\log_2 \frac{P(x)P(y \mid x)}{P(x)P(y)} \\
&= \sum_{x,y} P(x,y)\log_2 \frac{P(y \mid x)}{P(y)} \\
&= \sum_{x,y} P(x,y)\log_2 P(y \mid x) - \sum_{x,y} P(x,y)\log_2 P(y) \\
&= -H(Y \mid X) + H(Y)\\
\text{So, } I(X;Y) &= H(Y) - H(Y \mid X)
\end{align*}

Also,
\[
H(X,Y) = H(X) + H(Y \mid X)
\]

Therefore,
\[
H(Y \mid X) = H(X,Y) - H(X)
\]

Substituting this into the previous expression,
\[
I(X;Y) = H(Y) - H(X,Y) + H(X)
\]

Thus the mutual information formula becomes
\[
I(X;Y) = H(X) + H(Y) - H(X,Y)
\]

\subsection{Convexity and Jensen's Inequality}

\begin{definitionbox}[Convex Function]
A function $f$ is convex over an interval if its second derivative is non-negative over that interval:
\[
\frac{d^2 f(x)}{dx^2} \ge 0
\]
Examples: $x^2$ and $e^x$.
\end{definitionbox}

Consider the function $f(x)=x^2$ with $a=0$ and $b=2$.

\begin{itemize}
    \item $f(0)=0$ and $f(2)=4$
    \item Average of the function values:
\[
\frac{f(0)+f(2)}{2} = \frac{0+4}{2} = 2
\]
    \item Function of the average:
\[
f\left(\frac{a+b}{2}\right)
= f\left(\frac{0+2}{2}\right)
= f(1)
= 1^2
= 1
\]
\end{itemize}

\begin{infobox}[Key Observation]
For a convex function, the average of the function is greater than or equal to the function of the average:
\[
\text{Average of the function} \ge \text{Function of the average}
\]
\end{infobox}

\begin{figure}[H]
    \centering
    \begin{tikzpicture}[scale=1.0, >=stealth]
        \draw[->, thick] (0,0) -- (4.2,0) node[right] {$x$};
        \draw[->, thick] (0,0) -- (0,3.2) node[above] {$f(x)$};

        \draw[thick, color=chaptercolor] (0,0) -- (3,2.4);

        \draw[dashed] (1.25,0) -- (1.25,1);
        \draw[dashed] (0,1) -- (1.25,1);
        \draw[dashed] (2.5,0) -- (2.5,2);
        \draw[dashed] (0,2) -- (2.5,2);

        \filldraw (1.25,1) circle (1.6pt);
        \filldraw (2.5,2) circle (1.6pt);

        \node[below] at (1.25,0) {$1$};
        \node[below] at (2.5,0) {$2$};
        \node[left] at (0,1) {$1$};
        \node[left] at (0,2) {$2$};
    \end{tikzpicture}
    \caption{Graphical view of the average-of-function versus function-of-average idea}
\end{figure}

\begin{formulabox}[Jensen's Inequality]
If a function $f(x)$ is convex over an interval $(a,b)$, then for every $x_1,x_2 \in (a,b)$ and $0 \le \lambda \le 1$,
\[
\lambda f(x_1) + (1-\lambda)f(x_2) \ge f\bigl[\lambda x_1 + (1-\lambda)x_2\bigr]
\]
A function $f$ is said to be strictly convex if equality holds only when $\lambda = 0$ or $\lambda = 1$.
\end{formulabox}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.4\textwidth]{Figures/jensens-inequality.pdf}
    \label{fig:jensens-inequality}
    \caption{Geometric interpretation of Jensen's inequality for a convex function}
\end{figure}

\begin{questionanswerbox}[Example]
For $x_1=0$, $x_2=2$, $\lambda=0.5$, and $f(x)=x^2$,
\[
\lambda f(x_1) + (1-\lambda)f(x_2) \ge f\bigl[\lambda x_1 + (1-\lambda)x_2\bigr]
\]
\[
0.5\times 0 + 0.5\times 4 \ge f[0.5\times 0 + 0.5\times 2]
\]
\[
2 \ge 1
\]
Thus $f(x)=x^2$ is a convex function.
\end{questionanswerbox}

\begin{definitionbox}[Concave Function]
A function $f$ is concave over an interval if its second derivative is non-positive over that interval:
\[
\frac{d^2 f(x)}{dx^2} \le 0
\]
Examples: $\log x$ and $\sqrt{x}$.
\end{definitionbox}

\begin{infobox}[Relation with Jensen's Inequality]
\begin{itemize}
    \item For a concave function, Jensen's inequality reverses direction.
\end{itemize}
\[
\lambda f(x_1) + (1-\lambda)f(x_2) \le f\bigl[\lambda x_1 + (1-\lambda)x_2\bigr]
\]
\begin{itemize}
    \item So for a concave function:
\[
\text{Average of the function} \le \text{Function of the average}
\]
    \item This is the version used later in the Gibbs information inequality because $\log(x)$ is a concave function.
\end{itemize}
\end{infobox}

\subsection{Gibbs Information Inequality}

\begin{formulabox}[Gibbs Information Inequality]
Let $P(x)$ and $Q(x)$, where $x \in X$, be two probability mass functions. Then
\[
D(P \parallel Q) \ge 0
\]
with equality if and only if $P(x)=Q(x)$ for all $x$.
\end{formulabox}

\begin{infobox}[Beginner Meaning]
\begin{itemize}
    \item Gibbs information inequality says that relative entropy can never be negative.
    \item So, the ``distance-like'' mismatch between a true distribution $P$ and an assumed distribution $Q$ is always zero or positive.
    \item It becomes exactly zero only when the two distributions are the same at every point.
    \item This is why KL divergence is useful for measuring how wrong an estimate is.
\end{itemize}
\end{infobox}

Let
\[
A = \{x : P(x) > 0\}
\]
be the support set of $P(x)$. Then
\begin{infobox}[How to Read the Proof]
\begin{itemize}
    \item The set $A$ contains only the values where $P(x)$ is actually nonzero.
    \item We work on $A$ so the ratio $\dfrac{Q(x)}{P(x)}$ is well-defined inside the summation.
    \item The main trick is Jensen's inequality applied to the concave function $\log(\cdot)$.
    \item The proof shows that $-D(P\parallel Q)\le 0$, which is the same as saying $D(P\parallel Q)\ge 0$.
\end{itemize}
\end{infobox}

\begin{align*}
-D(P \parallel Q)
&= -\sum_{x\in A} P(x)\log \frac{P(x)}{Q(x)} = \sum_{x\in A} P(x)\log \frac{Q(x)}{P(x)} \\
\text{or, } -D(P \parallel Q) &\le \log \sum_{x\in A} P(x)\frac{Q(x)}{P(x)} = \log \sum_{x\in A} Q(x) \\
\text{or, } -D(P \parallel Q) &\le \log \sum_{x\in X} Q(x) = \log 1 = 0
\end{align*}

The inequality step follows because $\log(\cdot)$ is a concave function, so Jensen's inequality is used in its reversed form here.

\begin{infobox}[What Happened in the Middle Steps]
\begin{itemize}
    \item The third line is the Jensen step: average of the log is less than or equal to log of the average.
    \item The next line simplifies because $P(x)$ cancels with the denominator inside $\dfrac{Q(x)}{P(x)}$.
    \item Since $A \subseteq X$, the sum over $A$ cannot exceed the full sum over $X$.
    \item And because $Q(x)$ is a probability mass function, \(\sum_{x\in X}Q(x)=1\).
\end{itemize}
\end{infobox}

Therefore,
\[
D(P \parallel Q) \ge 0
\]
and relative entropy becomes zero only when the two distributions are identical.

\begin{infobox}[First-Timer Takeaway]
\begin{itemize}
    \item Think of $P(x)$ as the correct probability model and $Q(x)$ as the model we guessed.
    \item Gibbs information inequality says the penalty for using the wrong model can never be negative.
    \item If our guess is perfect, then the penalty is zero. If the guess is off, the penalty becomes positive.
    \item This is why KL divergence is often read as a measure of ``how different'' the assumed distribution is from the true one.
\end{itemize}
\end{infobox}

\subsection{Example: Asymmetric Nature of Relative Entropy}

\begin{questionanswerbox}[Example]
\textbf{Q:} Let $X=\{0,1\}$ and consider two distributions $P$ and $Q$ on $X$:
\[
P(0)=1-r,\qquad P(1)=r,\qquad Q(0)=1-s,\qquad Q(1)=s
\]
Find $D(P \parallel Q)$ and $D(Q \parallel P)$.

\textbf{A:} Given,
\begin{align*}
    X=\{0,1\}, \qquad P(0)&=1-r,\qquad P(1)=r\\
    Q(0)&=1-s,\qquad Q(1)=s
\end{align*}

We have,
\[
D(P \parallel Q)
= \sum_x P(x)\log_2 \frac{P(x)}{Q(x)}
= P(0)\log_2\frac{P(0)}{Q(0)} + P(1)\log_2\frac{P(1)}{Q(1)}
\]
\[
= (1-r)\log_2\frac{1-r}{1-s} + r\log_2\frac{r}{s}
\]

When $r=s=\dfrac{1}{2}$, then
\[
D(P \parallel Q)=0
\]

When $r=\dfrac{1}{2}$ and $s=\dfrac{1}{4}$,
\[
D(P \parallel Q)
= \frac{1}{2}\log_2\frac{1-\frac{1}{2}}{1-\frac{1}{4}}
+ \frac{1}{2}\log_2\frac{\frac{1}{2}}{\frac{1}{4}}
\]
\[
= \frac{1}{2}\log_2\frac{\frac{1}{2}}{\frac{3}{4}}
+ \frac{1}{2}\log_2\frac{\frac{1}{2}}{\frac{1}{4}}
\]
\[
= -0.292481 + \frac{1}{2}
= 0.2075 \text{ bit}
\]

Similarly,
\[
D(Q \parallel P)
= (1-s)\log_2\frac{1-s}{1-r} + s\log_2\frac{s}{r}
\]
\[
= \left(1-\frac{1}{4}\right)\log_2\frac{1-\frac{1}{4}}{1-\frac{1}{2}}
+ \frac{1}{4}\log_2\frac{\frac{1}{4}}{\frac{1}{2}}
\]
\[
= \frac{3}{4}\log_2\frac{\frac{3}{4}}{\frac{1}{2}}
+ \frac{1}{4}\log_2\frac{1}{2}
\]
\[
= 0.438721 - 0.25
\]
\[
= 0.188721 \text{ bit}
\]

Hence,
\[
D(P \parallel Q) \ne D(Q \parallel P)
\]
So the asymmetric relationship is proved.
\end{questionanswerbox}

\section{Chain Rule}

\subsection{Entropy}
\begin{formulabox}[Chain Rule for Entropy]
\[
H(X_1,X_2,\ldots,X_n) = \sum_{i=1}^{n} H(X_i \mid X_1,\ldots,X_{i-1})
\]
\end{formulabox}

\begin{infobox}[Simple Reading]
\begin{itemize}
    \item The total uncertainty is built step by step.
    \item Start with the uncertainty of $X_1$.
    \item Then add the extra uncertainty of $X_2$ after $X_1$ is already known.
    \item Then add the extra uncertainty of $X_3$ after $X_1$ and $X_2$ are known, and so on.
\end{itemize}
\end{infobox}

\begin{itemize}
    \item For $n=2$:
\[
H(1,2)=H(1)+H(2 \mid 1)
\]
    \item For $n=3$:
\[
H(1,2,3)=H(1)+H(2 \mid 1)+H(3 \mid 1,2)
\]
\end{itemize}

\subsection{Conditional Entropy}
\begin{formulabox}[Chain Rule for Conditional Entropy]
\[
H(X_1,X_2,\ldots,X_n \mid Y) = \sum_{i=1}^{n} H(X_i \mid X_1,\ldots,X_{i-1},Y)
\]
\end{formulabox}

\begin{infobox}[Simple Reading]
\begin{itemize}
    \item Here the total uncertainty is measured while $Y$ is already known.
    \item Start with the uncertainty of $X_1$ when $Y$ is known.
    \item Then add the extra uncertainty of $X_2$ after $X_1$ and $Y$ are known.
    \item Then add the extra uncertainty of $X_3$ after $X_1$, $X_2$, and $Y$ are known, and so on.
\end{itemize}
\end{infobox}

\begin{itemize}
    \item For $n=2$:
\[
H(X_1,X_2 \mid Y)=H(X_1 \mid Y)+H(X_2 \mid X_1,Y)
\]
    \item For $n=3$:
\[
H(X_1,X_2,X_3 \mid Y)=H(X_1 \mid Y)+H(X_2 \mid X_1,Y)+H(X_3 \mid X_1,X_2,Y)
\]
\end{itemize}

\subsection{Mutual Information}
\begin{formulabox}[Chain Rule for Mutual Information]
\[
I(X_1,X_2,\ldots,X_n;Y) = \sum_{i=1}^{n} I(X_i;Y \mid X_1,\ldots,X_{i-1})
\]
\end{formulabox}

\begin{infobox}[Why This Helps]
\begin{itemize}
    \item This says the information that the whole group gives about $Y$ can be counted one variable at a time.
    \item \textbf{Each new term tells us how much extra information the next variable adds after the previous ones are already known}.
    \item So the chain rule avoids double counting.
\end{itemize}
\end{infobox}

\begin{itemize}
    \item For $n=2$:
\[
I(1,2;Y)=I(1;Y)+I(2;Y \mid 1)
\]
    \item For $n=3$:
\[
I(1,2,3;Y)=I(1;Y)+I(2;Y \mid 1)+I(3;Y \mid 1,2)
\]
\end{itemize}

\section{Discrete Memoryless Channel}

\begin{definitionbox}[Discrete Memoryless Channel (DMC)]
A discrete memoryless channel (DMC) is a channel where:
\begin{itemize}
    \item the input and output alphabets are discrete (finite),
    \item the current output depends only on the current input,
    \item the current output does not depend on any previous inputs or outputs.
\end{itemize}
So, the channel has no memory.
\end{definitionbox}

\begin{definitionbox}[Channel Matrix]
The channel matrix of a DMC is a table showing the conditional probabilities of receiving output symbols given an input symbol.
\end{definitionbox}

\begin{questionanswerbox}[Example]

\]
then the channel matrix is
\[
\begin{bmatrix}
0.9 & 0.1 \\
0.2 & 0.8
\end{bmatrix}
\]
\end{questionanswerbox}

\section{Source Coding}

\begin{definitionbox}[Source Coding]
Source coding is the process of compressing data to reduce redundancy while preserving the essential information.
\end{definitionbox}

It is used to represent data efficiently for transmission or storage. The goal is to minimize the number of bits required to encode the source symbols without losing information.

\subsection{Types of Source Coding}

There are two types of source coding:

\subsubsection{Lossless Source Coding}

\begin{definitionbox}[Lossless Source Coding]
Ensures that the original data can be perfectly reconstructed from the compressed data without any loss of information.
\end{definitionbox}

\begin{itemize}
    \item Examples: Huffman coding, run-length coding, arithmetic coding
    \item Use case: text files, medical imaging
\end{itemize}

\subsubsection{Lossy Source Coding}

\begin{definitionbox}[Lossy Source Coding]
Reduces data size by discarding less important information, which may result in loss of quality.
\end{definitionbox}

\begin{itemize}
    \item Examples: JPEG (images), MP3 (audio), MPEG (video)
    \item Use case: multimedia applications where minor losses are acceptable
\end{itemize}

\subsection{Source Coding Theorem}

\begin{definitionbox}[Source Coding Theorem]
The source coding theorem introduced by Claude Shannon states that the average length \(L\) of the encoded message for a discrete memoryless source \(X\) cannot be less than the entropy \(H(X)\) of the source.
\end{definitionbox}

\begin{formulabox}[Source Coding Bound]
\[
L \ge H(X)
\]
\begin{itemize}
    \item \(L\) = average codeword length
    \item \(H(X)\) = entropy of the source
\end{itemize}
\end{formulabox}

Efficient encoding schemes like Huffman coding can achieve \(L \approx H(X)\), but cannot go below it. The theorem ensures optimal compression, minimizing redundancy in data transmission or storage.

\begin{formulabox}[Average Codeword Length]
\[
L = \sum P(x_i)\,l(x_i)
\]
\begin{itemize}
    \item \(L\) = average codeword length
    \item \(P(x_i)\) = probability of symbol \(x_i\)
    \item \(l(x_i)\) = length of the codeword assigned to symbol \(x_i\)
\end{itemize}
\end{formulabox}

\begin{formulabox}[Coding Efficiency]
\[
\eta = \frac{H(X)}{L}\times 100\%
\]
\begin{itemize}
    \item \(\eta\) = efficiency of the coding scheme
    \item \(H(X)\) = source entropy
    \item \(L\) = encoded average length. Length = Codeword $\times$ Probability
\end{itemize}
\end{formulabox}

Coding efficiency measures how well a source-coding scheme performs compared to the theoretical limit of compression given by the source entropy. For a perfectly efficient coding scheme,
\[
L = H(X)
\]

\begin{formulabox}[Redundancy]
\[
R_e = 1-\eta
\]
\begin{itemize}
    \item \(R_e\) = redundancy of the code
    \item Use this form when \(\eta\) is written as a fraction, for example \(\eta=1\)
    \item If efficiency is written in percent, then use \(R_e = 100\% - \eta\)
\end{itemize}
\end{formulabox}

Redundancy tells us how much extra coding length remains beyond the ideal limit. If efficiency is high, redundancy is low. For a perfect code, \(\eta=1\) or \(100\%\), so redundancy becomes \(0\).

\subsection{Kraft-McMillan Inequality}

\begin{formulabox}[Kraft-McMillan Inequality]
The Kraft-McMillan inequality provides a necessary and sufficient condition for the existence of a prefix-free (uniquely decodable) code. For a set of codewords with lengths \(n_1,n_2,\ldots,n_n\),
\[
\sum_{k=1}^{L} 2^{-n_k} \le 1
\]
\begin{itemize}
    \item \(n_k\) = length of the \(k\)-th codeword
    \item \(L\) = total number of codewords in the code
    \item \(2^{-n_k}\) = contribution of a codeword of length \(n_k\) in a binary code
    \item \(\sum_{k=1}^{L} 2^{-n_k}\) = total Kraft sum of all codewords
\end{itemize}
If the inequality is satisfied, a prefix-free code with those codeword lengths can be constructed. If the sum exceeds 1, such a code cannot exist. Equality implies an optimally efficient code.
\end{formulabox}

\subsubsection{What Prefix Means in Prefix-Free Code}

A code is prefix-free if no complete codeword is the prefix of another complete codeword.

\begin{infobox}[Simple Reading]
\begin{itemize}
    \item Having the same starting bits is allowed.
    \item The real problem starts when one full codeword appears at the beginning of another full codeword.
    \item So, we compare complete codewords, not just their first few bits.
\end{itemize}
\end{infobox}

Suppose we have:
\begin{itemize}
    \item \(000\) for \(F\)
    \item \(001\) for \(A\)
    \item \(010\) for \(D\)
    \item \(011\) for \(E\)
    \item \(10\) for \(B\)
    \item \(11\) for \(C\)
\end{itemize}

Now check:
\begin{itemize}
    \item Is \(000\) a prefix of \(001\)? No. They are both complete 3-bit codewords, and one does not fully contain the other.
    \item Does any complete codeword appear at the start of a longer codeword? No.
\end{itemize}

So this is a prefix-free code.

\paragraph{Example of a Bad Code}
Suppose:
\begin{itemize}
    \item \(A \to 0\)
    \item \(B \to 01\)
\end{itemize}

Here, \(0\) is a prefix of \(01\). So this is not prefix-free.

\begin{questionanswerbox}[Prefix Code Condition Example]
\textbf{Q:} Discuss the possibility of finding a prefix code with codeword lengths \(1,2,3,3\).

\textbf{A:}

\textbf{Given,}
\[

\]

\begin{align*}
\sum_{k=1}^{L} 2^{-n_k}
&= 2^{-1}+2^{-2}+2^{-3}+2^{-3}\\
&= 1 \le 1
\end{align*}

Hence, the Kraft inequality is satisfied. So the prefix-free code is possible.

\subsection{Huffman Coding}

\begin{definitionbox}[Huffman Coding]
Huffman coding is a lossless data-compression algorithm that assigns shorter binary codes to more frequent symbols and longer codes to less frequent ones, thereby minimizing the average code length.
\end{definitionbox}

It is generally useful for data with frequently occurring characters.

\subsubsection{Steps / Algorithm}
\begin{enumerate}
    \item Arrange all the symbols with their probabilities in ascending order.
    \item Combine the two least probable symbols into a node, summing their probabilities.
    \item Repeat until a single tree (Huffman tree) is formed.
    \item Assign binary codes: \(0\) to the left branch and \(1\) to the right branch.
\end{enumerate}

\section{Arithmetic Coding}

\begin{definitionbox}[Arithmetic Coding]
Arithmetic coding is a lossless data-compression technique that represents an entire message as a single number within the interval \([0,1)\).
\end{definitionbox}

Unlike traditional methods, it assigns fractional ranges to symbols based on their probabilities, enabling more efficient encoding for data with uneven symbol distributions.

\subsection{Efficiency of Arithmetic Coding over Huffman Coding}
\begin{itemize}
    \item \textbf{Precision:} Arithmetic coding uses actual probabilities, avoiding the rounding errors of Huffman coding.
    \item \textbf{Adaptability:} It encodes the entire message as a single fractional number, optimizing compression for skewed distributions.
    \item \textbf{Variable Length:} It dynamically adjusts encoding based on the input, reducing redundancy.
    \item \textbf{Handling Long Messages:} It considers the overall probability distribution and can achieve higher efficiency for long sequences.
\end{itemize}

Arithmetic coding can use codes of any length, which makes the encoding more precise and often more efficient than Huffman coding. However, it is more computationally complex and more sensitive to errors.

\begin{infobox}[First-Timer Idea]
\begin{itemize}
    \item Arithmetic coding does not assign one fixed binary codeword to each symbol.
    \item Instead, it keeps shrinking an interval according to the incoming symbols.
    \item At the end, any number inside the final interval can be used as the tag for the whole message.
\end{itemize}
\end{infobox}

\begin{formulabox}[Interval Update Rule]
\[
\text{Upper limit} = \text{Lower limit} + \text{Difference} \times \text{Probability}
\]
\begin{itemize}
    \item \textbf{Lower limit} = the current starting point of the interval
    \item \textbf{Difference} = current interval width
    \item \textbf{Probability} = probability of the symbol used to split the interval
\end{itemize}
\end{formulabox}

\section{Shannon-Fano Coding}

\begin{definitionbox}[Shannon-Fano Coding]
Shannon-Fano coding is a lossless data-compression technique that assigns binary codes to symbols based on their probabilities.
\end{definitionbox}

The goal is to create prefix-free codes with minimal average length.

\subsection{Steps}
\begin{enumerate}
    \item Arrange the symbols in decreasing order of probability.
    \item Split the list into two parts with nearly equal total probability.
    \item Assign \(0\) to the first half and \(1\) to the second half.
    \item Recursively apply the process to each half until individual symbols are assigned codes.
\end{enumerate}

\begin{infobox}[First-Timer Idea]
\begin{itemize}
    \item Huffman coding builds a tree by combining the smallest probabilities.
    \item Shannon-Fano coding starts from the top and repeatedly splits the list into two nearly equal-probability groups.
    \item Both are lossless, but Huffman coding is usually more optimal.
\end{itemize}
\end{infobox}

\section{Lempel-Ziv Coding}

\begin{definitionbox}[Lempel-Ziv Encoding]
Lempel-Ziv encoding is a lossless data-compression method that replaces repeated occurrences of data by references to dictionary entries instead of writing the same pattern again and again.
\end{definitionbox}

It works by building a dictionary of substrings during encoding. Used in formats like GIF, ZIP, PNG and in PDF or some image files.

\subsection{Steps}
\begin{enumerate}
    \item Start with an empty or initialized dictionary.
    \item Read characters and check whether the current substring already exists in the dictionary.
    \item If it is not found, add the new substring to the dictionary and output the index of the previous known substring.
    \item Continue until the whole input message is processed.
\end{enumerate}

\section{Arithmetic Decoding}

\begin{definitionbox}[Arithmetic Decoding]
Arithmetic decoding is the reverse process of arithmetic coding. Starting from the tag value, we repeatedly find which probability interval contains the tag and then zoom into that interval until the original message is recovered.
\end{definitionbox}

\subsection{Steps}
\begin{enumerate}
    \item Construct the arithmetic-coded diagram using the given symbol probabilities.
    \item Find the symbol whose interval contains the tag value.
    \item Backtrace or renormalize inside that interval and continue decoding.
\end{enumerate}

\begin{infobox}[First-Timer Idea]
\begin{itemize}
    \item In arithmetic encoding, many symbols shrink one interval again and again.
    \item In arithmetic decoding, we do the opposite: we see where the tag sits, identify the symbol, then zoom into that smaller interval.
    \item Repeat this until the whole message is obtained.
\end{itemize}
\end{infobox}

\begin{questionanswerbox}[Bits Saved by Huffman Coding Example]
\textbf{Q:} A networking company uses a compression technique before transmission. The message contains the characters
\[
a\to 5,\quad b\to 9,\quad c\to 12,\quad d\to 13,\quad e\to 16,\quad f\to 45
\]
and each character in the original message takes \(1\) byte. If Huffman coding is used, how many bits are saved?

\textbf{A:}

\textbf{Given,}
\begin{align*}
\text{Total frequency} &= 5+9+12+13+16+45 = 100
\end{align*}

\begin{align*}
P(a)&=\frac{5}{100}=0.05, &
P(b)&=\frac{9}{100}=0.09,\\
P(c)&=\frac{12}{100}=0.12, &
P(d)&=\frac{13}{100}=0.13,\\
P(e)&=\frac{16}{100}=0.16, &
P(f)&=\frac{45}{100}=0.45
\end{align*}

\begin{align*}
\text{Total uncompressed bits} &= 1\text{ byte}\times 100 = 8 \text{ bit}\times 100 = 800 \text{ bits}
\end{align*}

    \centering
    \includegraphics[width=0.9\linewidth]{Figures/generated/huffman_bits_saved_networking_example.png}

\begin{tabular}{|C{1.4cm}|C{1.9cm}|C{2cm}|C{2.2cm}|C{1.4cm}|C{2.2cm}|}
    \hline
    \textbf{Symbol} & \textbf{Frequency} & \textbf{Probability} & \textbf{Codeword} & \textbf{Length} & \textbf{Compressed Bits} \\ \hline
    \(f\) & \(45\) & \(0.45\) & \(0\)    & \(1\) & \(45\times 1=45\) \\ \hline
    \(e\) & \(16\) & \(0.16\) & \(10\)   & \(2\) & \(16\times 2=32\) \\ \hline

\text{Total compressed bits} &= 45+32+52+48+36+20 = 233 \text{ bits}

\section{Run Length Encoding}

\begin{definitionbox}[Run Length Encoding]
Run Length Encoding (RLE) is a simple lossless compression technique that reduces repetitive data by storing consecutive repeated symbols as a count together with the symbol value.
\end{definitionbox}

\subsection{Steps}
\begin{enumerate}
    \item Read the input data sequentially.
    \item Count consecutive repetitions of the same symbol.
    \item Replace the repeated sequence by a pair like \((\text{count}, \text{symbol})\).
    \item If flags are used, add the flag before the repetitive symbol.
    \item Remove trivial single counts if the chosen format requires it.
\end{enumerate}

\begin{questionanswerbox}[Run Length Encoding Example]
\textbf{Q:} Compress the text \texttt{aaaa bbbbbb c ddddd} using run length encoding.

\textbf{A:}

\textbf{Given,}
\[
\texttt{aaaa\ bbbbbb\ c\ ddddd}
\]

The code is \(16\) bytes of data and the repetitions are:
\begin{itemize}
    \item \(a\): \(4\) times
    \item \(b\): \(6\) times
    \item \(c\): \(1\) time
    \item \(d\): \(5\) times
\end{itemize}

\textbf{Count-character pairs}
\[
(4,a),\ (6,b),\ (1,c),\ (5,d)
\]

\textbf{Using ASCII values,}
\begin{align*}
a &\to (4,a)\to (4,97)\\
b &\to (6,b)\to (6,98)\\
c &\to (1,c)\to (1,99)\\
d &\to (5,d)\to (5,100)
\end{align*}

\textbf{Stored byte sequence without a flag}
\[
04\ 97\ 06\ 98\ 01\ 99\ 05\ 100
\]
So the compressed data becomes \(8\) bytes.

\textbf{Flag-based form shown in the note}
\[
255\ 04\ 97\quad 255\ 06\ 98\quad 99\quad 255\ 05\ 100
\]

Here \(255\) is used as a flag before repeated runs, while the single symbol \(c\) is written directly as \(99\).

\textbf{Simple Reading}
\begin{itemize}
    \item RLE works well when the same symbol repeats many times.
    \item A run is stored as \((\text{count}, \text{symbol})\).
    \item If a symbol appears only once, RLE may not help much unless a smarter flag rule is used.
\end{itemize}
\end{questionanswerbox}

\begin{questionanswerbox}[Run Length Encoding for Mixed Data Example]
\textbf{Q:} Show why RLE may become inefficient when the data contains many single symbols.

\textbf{A:}

The lower half of the note gives a mixed string with:
\begin{itemize}
    \item \(10\) consecutive \(a\)'s,
    \item \(6\) consecutive \(b\)'s,
    \item many alternating single \(e\) and \(c\) symbols,
    \item \(15\) consecutive \(d\)'s,
    \item then a few more single symbols at the end.
\end{itemize}

So the RLE count-character form starts like:
\[
(10,a),\ (6,b),\ (1,e),\ (1,c),\ (1,e),\ (1,c),\ \ldots,\ (15,d),\ (1,e),\ (1,c),\ (1,b)
\]

\textbf{What the note is showing}
\begin{itemize}
    \item Long runs such as \(a^{10}\), \(b^6\), and \(d^{15}\) compress well.
    \item Alternating symbols like \(e,c,e,c,\ldots\) are mostly single-count runs.
    \item If every single symbol is still stored as \((1,\text{symbol})\), the encoded stream becomes large.
\end{itemize}

The note therefore compares two ideas:
\begin{itemize}
    \item \textbf{No-flag RLE:} even single symbols carry their count, so the representation becomes bulky.
    \item \textbf{Flag-value RLE:} only repeated runs are marked with a flag, so single symbols can be written directly and the result becomes shorter.
\end{itemize}

This is the key lesson of the page: RLE is effective for long repetitions, but it can perform poorly on data with many isolated symbols.
\end{questionanswerbox}

\section{Effective and Prefix-Free Codes}

\begin{questionanswerbox}[Effective Codes Example]
\textbf{Q:} What are effective codes?

\textbf{A:}
Effective codes are codes that represent information efficiently using the least possible number of bits while keeping communication reliable.

They are designed to:
\begin{itemize}
    \item maximize information carried per bit,
    \item reduce redundancy,
    \item use bandwidth and memory efficiently,
    \item maintain a balance between compression, error detection, and error correction.
\end{itemize}

The note also points out that the effectiveness of a code is often judged by:
\begin{itemize}
    \item how close its average length is to source entropy,
    \item and how reliably it works over a noisy channel.
\end{itemize}
\end{questionanswerbox}

\begin{questionanswerbox}[Prefix-Free Code Properties Example]
\textbf{Q:} What is a prefix-free code? Write its properties.

\textbf{A:}
A prefix-free code is a code in which no codeword is the prefix of any other codeword. This allows each codeword to be identified as soon as it ends.

\textbf{Properties:}
\begin{itemize}
    \item Uniquely decodable: each codeword can be decoded without ambiguity.
    \item Instantaneous decoding: symbols can be decoded as soon as the codeword ends.
    \item Satisfies Kraft's inequality:
    \[
    \sum 2^{-n_k}\le 1
    \]
    where \(n_k\) is the length of a codeword.
    \item Efficient for variable-length coding, such as Huffman coding.
\end{itemize}
\end{questionanswerbox}

\section{Additional Coding Concepts}

\begin{questionanswerbox}[Kraft's Inequality Example]
\textbf{Q:} Explain Kraft's inequality.

\textbf{A:}
Kraft's inequality gives a necessary and sufficient condition for the existence of a prefix-free code for a given set of codeword lengths.

For a \(D\)-ary code, usually binary with \(D=2\), the inequality is:
\[
\sum_{k=1}^{m} D^{-n_k}\le 1
\]

\textbf{Here,}
\begin{itemize}
    \item \(n_k\) = length of the \(k\)-th codeword
    \item \(m\) = total number of codewords
    \item \(D\) = size of the code alphabet
\end{itemize}

\textbf{Significance:}
\begin{itemize}
    \item If the inequality holds, a prefix-free code exists for those lengths.
    \item If it does not hold, such a prefix-free code cannot be constructed.
\end{itemize}
\end{questionanswerbox}

\begin{questionanswerbox}[Critical Rate and Majority Decoding Example]
\textbf{Q:} Define critical rate and majority decoding.

\textbf{A:}

\textbf{Critical rate:}
Critical rate refers to the maximum rate below which a coding scheme can achieve arbitrarily low probability of error as the code length tends to infinity.

It separates:
\begin{itemize}
    \item the region where reliable communication is possible,
    \item from the region where error probability remains bounded away from zero.
\end{itemize}

\textbf{Majority decoding:}
Majority decoding is an error-correction method in which each bit is decided from the majority of its repeated received values.

\textbf{Example:}
If a bit \(1\) is sent using repetition code as \(\texttt{1110}\), the receiver checks the majority value and decodes it as \(1\).

The note states that this method can correct one error in repeated transmission.
\end{questionanswerbox}
